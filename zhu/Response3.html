1. Addressing the Use of Increment/Decrement Operators in Array Access

"When I’m referring to the post-increment/post-decrement/pre-increment/pre-decrement usage, I’m referring to the usage within an array access, that is, inside the square brackets [] after the “parents” array field. The post/pre-increment/decrement operators are often used inside “for” loops and as standalone lines of code, but what I’m referring to as unusual is the embedded usage within an array access, which is unusual and hard to read. So I’m not saying simply using ++ or -- is unusual since that is typical usage, but using these embedded inside an array access is considered a confusing coding style since the array index is modified along with accessing the array."
-Dr. Zhu

I want to reiterate that Java and C have near same syntax but not the same programming principles and distance from bare metal.

I acknowledge that embedded usage of increment and decrement operators within array access may seem uncommon for some students. However, this practice is not confusing for students who are familiar with C programming. Both Java and C have similar syntax for increment and decrement operators and fixed array sizes, making this operation valid and readable with proper commenting.

In both Java and C, the increment (++) and decrement (--) operators can be used in postfix or prefix forms. The postfix form evaluates to the value before the increment/decrement, while the prefix form evaluates to the value after the increment/decrement. This behavior is consistent in both languages, making the syntax intuitive for those familiar with one language to understand the other.

The point of the loop, which Dr. Zhu is refers to as"modifying the array index due to embedded usage within an array access" is essential to the business logic.

Dr. Zhu's student samples have also modified the array index. It is a common practice in C programming, especially when traversing trees or linked lists. The increment and decrement operators are used to move the index to the next or previous element in the array, respectively.

void printDonutPattern(int arr[], int size) {

  for (int k = 0; k < size; k++) printf("%d ", arr[i++]);

import java.util.Scanner;

public class BankingSystem {
  private static Account[] accounts = new Account[1]; // Start with an array of size 1
  private static int numAccounts = 0;

  public static class Account {
    private int accN;
    private String name;
    private double balance;

    public Account(int accN, String name, double balance) {
      this.accN = accN;
      this.name = name;
      this.balance = balance;
    }

    public int getAccountNumber() { return accN; }
    public String getAccountHolderName() { return name; }
    public double getBalance() { return balance; }

    public void deposit(double amount) {
      if (amount > 0) balance += amount;
      else System.out.println("Deposit amount must be positive.");
    }

    public void withdraw(double amount) {
      if (balance >= amount) balance -= amount;
      else System.out.println("Insufficient funds");
    }

    @Override
    public String toString() {
      return "Account Number: " + accN + ", Name: " + name + ", Balance: " + balance;
    }
  }

  private static void addAccount(int accN, String name, double balance) {
    if (numAccounts == accounts.length) {
      Account[] newAccounts = new Account[accounts.length * 2];
      for (int i = 0; i < accounts.length; i++) newAccounts[i] = accounts[i];
      accounts = newAccounts;
    }
    accounts[numAccounts++] = new Account(accN, name, balance);
    System.out.println("Account added: " + name + " with balance " + balance);
  }

  private static void display() {
    for (int i = 0; i < numAccounts; i++) System.out.println(accounts[i]);
  }

  private static void bubbleSort() {
    for (int i = 0; i < numAccounts - 1; i++) {
      for (int j = 0; j < numAccounts - i - 1; j++) {
        if (accounts[j].getAccountNumber() > accounts[j + 1].getAccountNumber()) {
          Account temp = accounts[j];
          accounts[j] = accounts[j + 1];
          accounts[j + 1] = temp;
        }
      }
    }
  }

  private static Account binarySearch(int accN) {
    int left = 0;
    int right = numAccounts - 1;
    while (left <= right) {
      int mid = left + (right - left) / 2;
      if (accounts[mid].getAccountNumber() == accN) return accounts[mid];
      if (accounts[mid].getAccountNumber() < accN) left = mid + 1;
      else right = mid - 1;
    }
    return null;
  }

  public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);
    System.out.print("Enter the number of customers to add: ");
    int numCustomers = scanner.nextInt();
    scanner.nextLine(); // consume newline
    for (int i = 0; i < numCustomers; i++) {
      System.out.print("Enter account number for customer " + (i + 1) + ": ");
      int accN = scanner.nextInt();
      scanner.nextLine(); // consume newline
      System.out.print("Enter account holder name for customer " + (i + 1) + ": ");
      String name = scanner.nextLine();
      System.out.print("Enter initial balance for customer " + (i + 1) + ": ");
      double balance = scanner.nextDouble();
      scanner.nextLine(); // consume newline
      addAccount(accN, name, balance);
    }

    boolean exit = false;
    while (!exit) {
      System.out.println("\nSelect an option:");
      System.out.println("1. Display all account details");
      System.out.println("2. Search by account number");
      System.out.println("3. Deposit amount");
      System.out.println("4. Withdraw amount");
      System.out.println("5. Exit");
      int choice = scanner.nextInt();
      scanner.nextLine(); // consume newline
      switch (choice) {
        case 1:
          display();
          break;
        case 2:
          System.out.print("Enter account number to search: ");
          int accNumber = scanner.nextInt();
          scanner.nextLine(); // consume newline
          Account acc = binarySearch(accNumber);
          if (acc != null) System.out.println(acc);
          else System.out.println("Account not found");
          break;
        case 3:
          System.out.print("Enter account number to deposit amount: ");
          int accountNumDeposit = scanner.nextInt();
          scanner.nextLine(); // consume newline
          Account accDeposit = binarySearch(accountNumDeposit);
          if (accDeposit != null) {
            System.out.print("Enter amount to deposit: ");
            double amountDeposit = scanner.nextDouble();
            accDeposit.deposit(amountDeposit);
            System.out.println("Amount deposited successfully");
          } else System.out.println("Account not found");
          break;
        case 4:
          System.out.print("Enter account number to withdraw amount: ");
          int accountNumWithdraw = scanner.nextInt();
          scanner.nextLine(); // consume newline
          Account accWithdraw = binarySearch(accountNumWithdraw);
          if (accWithdraw != null) {
            System.out.print("Enter amount to withdraw: ");
            double amountWithdraw = scanner.nextDouble();
            accWithdraw.withdraw(amountWithdraw);
            System.out.println("Amount withdrawn successfully");
          } else System.out.println("Account not found");
          break;
        case 5:
          exit = true;
          break;
        default:
          System.out.println("Invalid choice. Please try again.");
          break;
    }
    }
    scanner.close();
  }
}
